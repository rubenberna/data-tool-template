```bash
npm install --save data-tool-template
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'data-tool-template'
import 'data-tool-template/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
